011package de.dentrassi.maven.osgi.dp;
012
013import java.io.File;
014import java.io.IOException;
015import java.nio.file.Path;
016import java.util.Map;
017import java.util.jar.Manifest;
018
019import org.apache.maven.artifact.Artifact;
020import org.apache.maven.plugin.MojoExecutionException;
021import org.apache.maven.plugins.annotations.LifecyclePhase;
022import org.apache.maven.plugins.annotations.Mojo;
023import org.apache.maven.plugins.annotations.ResolutionScope;
024
025/**
026 * Build this project as an OSGi distribution package.
027 * <p>
028 * <strong>Note:</strong> Use the packaging type <code>dp</code> instead of
029 * directly configuring this mojo.
030 * </p>
031 * <p>
032 * Be sure to add this plugin as an extension:
033 * </p>
034 * {@code <plugins>
035    <plugin>
036        <groupId>de.dentrassi.maven</groupId>
037        <artifactId>osgi-dp</artifactId>
038        <extensions>true</extensions>
039    </plugin>
040</plugins>}
041 * <p>
042 * This project takes the dependencies of this project and builds an OSGi DP
043 * from it.
044 * </p>
045 *
046 * @author Jens Reimann
047 * @since 0.3.0
048 */
049@Mojo(name = "package", defaultPhase = LifecyclePhase.PACKAGE, requiresProject = true, requiresDependencyResolution = ResolutionScope.RUNTIME, requiresDependencyCollection = ResolutionScope.RUNTIME, threadSafe = false)
050public class PackageMojo extends AbstractDpMojo {
051
052    @Override
053    protected void attach(final Path out) {
054        this.project.getArtifact().setFile(out.toFile());
055    }
056
057    @Override
058    protected void fillFromDependencies(final Manifest manifest, final Map<String, File> files)
059            throws IOException, MojoExecutionException {
060
061        super.fillFromDependencies(manifest, files);
062
063        for (final Artifact art : this.project.getDependencyArtifacts()) {
064            processArtifact(manifest, files, art.getFile());
065        }
066    }
067}
